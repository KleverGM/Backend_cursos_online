# Comandos para ejecutar en tu servidor Azure:

# 1. Verificar conexión a MongoDB (usa mongosh en vez de mongo)
mongosh --eval "db.adminCommand('ping')"

# 2. Ver las notificaciones actuales en MongoDB
mongosh mongo_cursos_online --eval "db.notificaciones.find().limit(5)"

# 3. Verificar variables de entorno de Gunicorn
sudo cat /etc/systemd/system/gunicorn.service

# 4. Ver STDOUT de Gunicorn (aquí deberían aparecer los prints)
sudo journalctl -u gunicorn --no-pager -n 100
